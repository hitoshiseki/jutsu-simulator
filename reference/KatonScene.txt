#import "KatonScene.h"
#import "MenuScene.h"
#import "SwipeInstructionsScene.h"

@implementation KatonScene

+(KatonScene *) scene {
    return [[self alloc] init];
}

-(id) init {
    NSString *oImagePath = [[NSBundle mainBundle] pathForResource:@"particle_base" ofType:@"png"];
    NSString *dImagePath = [NSHomeDirectory() stringByAppendingString:@"/Documents/particle_base.png"];
    NSError *error = nil;
    [[NSFileManager defaultManager] copyItemAtPath:oImagePath toPath:dImagePath error:&error];
    NSLog(@"%@", error);
    NSMutableDictionary *dic = [NSMutableDictionary dictionaryWithContentsOfFile:[[NSBundle mainBundle] pathForResource:@"KatonTeste" ofType:@"plist"]];
    NSString *path = [NSHomeDirectory() stringByAppendingFormat:@"/Documents/KatonTeste.plist"];
    NSMutableDictionary *dic2 = [NSMutableDictionary dictionaryWithContentsOfFile:path];
    
    CGSize winSize = [[CCDirector sharedDirector] viewSize];
    NSLog(@"%f", winSize.width/2);
    dic2 = dic;
    [dic2 setObject:[NSString stringWithFormat:@"%f", winSize.width/2] forKey:@"sourcePositionx"];
    [dic2 writeToFile:path atomically:YES];
    
    self = [super initWithParticle:[CCParticleSystem particleWithFile:path] andType:KATON];
    particleDictionary = dic2;
    if (!self) return nil;

    return self;
}

-(void) changeParticleToStandBy {
    if (particleState != kPARTICLE_STATE_STANDBY) {
        particleState = kPARTICLE_STATE_STANDBY;
        mainParticle.speed = 0.0f;
        mainParticle.speedVar = 0.0f;
        mainParticle.gravity = CGPointMake(0.0f, 0.0f);
        mainParticle.tangentialAccelVar = 0.0f;
        mainParticle.startSize = 20.0f;
        mainParticle.startSizeVar = 0.0f;
        mainParticle.endSize = 0.0f;
        mainParticle.endSizeVar = 0.0f;
        [[OALSimpleAudio sharedInstance].backgroundTrack fadeTo:0.0f duration:1.2f target:self selector:@selector(stopAllSounds)];
    }
}

-(void) stopAllSounds {
    [Util stopAllSounds];
    [[OALSimpleAudio sharedInstance].backgroundTrack setVolume:1.0f];
}

-(void) changeParticleToActive {
    if (particleState != kPARTICLE_STATE_ACTIVE) {
        particleState = kPARTICLE_STATE_ACTIVE;
        mainParticle.life = [[particleDictionary objectForKey:@"particleLifespan"] floatValue];
        mainParticle.totalParticles = [[particleDictionary objectForKey:@"maxParticles"] floatValue];
        mainParticle.speed = [[particleDictionary objectForKey:@"speed"] floatValue];
        mainParticle.speedVar = [[particleDictionary objectForKey:@"speedVar"] floatValue];
        mainParticle.gravity = CGPointMake([[particleDictionary objectForKey:@"gravityx"] floatValue], [[particleDictionary objectForKey:@"gravityy"] floatValue]);
        mainParticle.tangentialAccelVar = [[particleDictionary objectForKey:@"tangentialAccelVariance"] floatValue];
        mainParticle.startSize = [[particleDictionary objectForKey:@"startParticleSize"] floatValue];
        mainParticle.startSizeVar = [[particleDictionary objectForKey:@"startParticleSizeVariance"] floatValue];
        mainParticle.endSize = [[particleDictionary objectForKey:@"finishParticleSize"] floatValue];
        mainParticle.endSizeVar = [[particleDictionary objectForKey:@"startParticleSizeVariance"] floatValue];
        [mainParticle resetSystem];
        [Util playHitSoundFor:KATON];
    }
}


@end
